   Guild[n] -&gt; {Equation -&gt; g,

g[i_] := (r[Subscript[x, i]] - 
     Sum[a[Subscript[x, i], Subscript[x, j]]*Subscript[n, j][t], {j, 
       Subscript[\[ScriptCapitalN], n]}])*Subscript[n, i][t];
a[xi_, xj_] := E^(-1/2*(xi - xj)^2/\[Sigma]^2);
r[x_] := 1 - x^2;
